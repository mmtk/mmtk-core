13/// This trait provides methods used during a trace.  The most important method is
14/// [`Self::trace_object`] which provides the way to trace an object during the current trace.  Many
15/// work packets depend on this trait to trace objects.
16///
17/// We need different implementations of this trait for the different behaviors in
18///
19/// -   different plans
20/// -   different traces of the same plan (e.g. marking trace and forwarding trace; nursery GC and
21///     mature GC; fast GC and defrag GC; etc.), and
22/// -   pinning (transitive or not) roots and regular edges.
23///
24/// Therefore, each GC selects one [`GCWorkContext`], and each [`GCWorkContext`] selects two
25/// [`Trace`] implementations for default edges and pinning edges, respectively.
26///
27/// A type of this trait shall be stateless and immutable.  It shall be cheap to instantiate from an
28/// [`MMTK`] instance.
29///
30/// This trait requires the [`Clone`] trait, and cloned instances behave exactly the same.
31///
32/// [`GCWorkContext`]: crate::scheduler::work::GCWorkContext
33pub trait Trace: 'static + Send + Clone {
34    /// The VM binding type this type serves.
35    type VM: VMBinding;
36
37    /// Instantiate from an [`MMTK`] instance.
38    ///
39    /// Note that values of this trait are usually instantiated in work packets that use it. It
40    /// should be reasonably cheap to instantiate.  Most types (such as [`PlanTrace`] and
41    /// [`GenNurseryTrace`]) only need a reference to the plan which can be downcasted from
42    /// [`MMTK::plan`].
43    ///
44    /// [`GenNurseryTrace`]: crate::plan::generational::gc_work::GenNurseryTrace
45    fn from_mmtk(mmtk: &'static MMTK<Self::VM>) -> Self;
46
47    /// Trace the `object`.  More precisely, it visits an object graph *edge* that points to
48    /// `object`.
49    ///
50    /// It may add `object` (or the new [`ObjectReference`] for `object` in a moving GC) into the
51    /// `queue`, which means its children needs to be recursively traversed.
52    ///
53    /// In non-moving GC, the return value is always `object`.  In moving GC, the return value may
54    /// be `object` or the new [`ObjectReference`] for the `object`.  If the return value is not
55    /// `object`, the slot that represents the object graph edge needs to be updated to hold the
56    /// return value instead.
57    ///
58    /// Its implementation generally needs to figure out which space an object resides in, and
59    /// invoke the right "trace object" method of the space for the current trace.
60    ///
61    /// # Notes
62    ///
63    /// ## The enqueued value and the return value
64    ///
65    /// The return value may be different from the enqueued value.  For example, during the
66    /// forwarding stage of MarkCompact, it returns the new object reference, but enqueues the old
67    /// `object` because the object has not been moved, yet.
68    ///
69    /// ## The `queue` can be a callback instead of a collection
70    ///
71    /// `FnMut(ObjectReference)` implements the [`ObjectQueue`] trait.  This means you can use a
72    /// lambda expression at the place of the `queue` argument.  For example, you can scan the
73    /// object immediately instead of adding the object reference into a container.
74    ///
75    /// Example:
76    ///
77    /// ```rust
78    /// trace.trace_object(worker, object, &mut |enqueued_object| {
79    ///     // Process the enqueued_object here...
80    /// });
81    /// ```
82    fn trace_object<Q: ObjectQueue>(
83        &self,
84        worker: &mut GCWorker<Self::VM>,
85        object: ObjectReference,
86        queue: &mut Q,
87    ) -> ObjectReference;
88
89    /// The post-scan hook to be call after scanning `object`.
90    ///
91    /// Each object is scanned by [`Scanning::scan_object`] or
92    /// [`Scanning::scan_object_and_trace_edges`], and this function will be called after scanning
93    /// an object as a hook to invoke possible policy-specific post-scan methods.  If `object` is in
94    /// a space that needs such a hook, this method should call such hook of the space.  Otherwise,
95    /// this method may do nothing.
96    ///
97    /// Currently, only [`ImmixSpace`] needs this hook to mark the line.
98    ///
99    /// [`ImmixSpace`]: crate::policy::immix::ImmixSpace
100    fn post_scan_object(&self, object: ObjectReference);
101
102    /// Return `true` if [`Self::trace_object`] may move any object.  If any space of the current
103    /// plan may move objects during this trace, this method should return `true`.
104    ///
105    /// If it returns `false`, it means [`Self::trace_object`] is guaranteed not to move the
106    /// `object`, and it is safe to elide updating a slot after tracing the reference in the slot.
107    /// It is always correct to conservatively return `true`.
108    ///
109    /// Note that this method is called very frequently, so it must be efficient.
110    fn may_move_objects() -> bool;
111}

114pub type SlotOfTrace<T> = <<T as Trace>::VM as VMBinding>::VMSlot;

116/// A [`Trace`] implementation that dispatches the `trace_object` method through
117/// [`crate::policy::sft::SFT::sft_trace_object`] using the Space Function Table (SFT).
118///
119/// Because SFT methods cannot be general, `SFTTrace` cannot be used for plans that needs multiple
120/// traces.  It is sufficient for simple plans such as `MarkSweep` and `SemiSpace`.
121#[allow(dead_code)]
122#[derive(Default)]
123pub struct SFTTrace<VM: VMBinding> {
124    phantom_data: PhantomData<VM>,
125}
126
127impl<VM: VMBinding> Clone for SFTTrace<VM> {
128    fn clone(&self) -> Self {
129        Self {
130            phantom_data: PhantomData,
131        }
132    }
133}
134
135impl<VM: VMBinding> Trace for SFTTrace<VM> {
136    type VM = VM;
137
138    fn from_mmtk(_mmtk: &'static MMTK<Self::VM>) -> Self {
139        Default::default()
140    }
141
142    fn trace_object<Q: ObjectQueue>(
143        &self,
144        worker: &mut GCWorker<VM>,
145        object: ObjectReference,
146        queue: &mut Q,
147    ) -> ObjectReference {
148        use crate::policy::sft::GCWorkerMutRef;
149
150        // Erase <VM> type parameter
151        let worker = GCWorkerMutRef::new(worker);
152
153        // Invoke trace object on sft
154        let sft = unsafe { crate::mmtk::SFT_MAP.get_unchecked(object.to_raw_address()) };
155
156        // Because `sft.sft_trace_object` cannot have generic parameters, we can't pass `queue`
157        // directly to it.  Instead we let `sft_trace_object` enqueue to this `tmp_queue` and
158        // forward the enqueued object to `queue`.
159        let mut tmp_queue = None;
160        let result = sft.sft_trace_object(&mut tmp_queue, object, worker);
161        if let Some(queued_object) = tmp_queue {
162            queue.enqueue(queued_object);
163        }
164        result
165    }
166
167    fn post_scan_object(&self, _object: ObjectReference) {
168        // Do nothing.  SFTTrace is only suitable for plans that don't need post_scan_object.
169    }
170
171    fn may_move_objects() -> bool {
172        // We conservatively assume it may move objects.
173        true
174    }
175}

177/// A [`Trace`] implementation that dispatches the `trace_object` method through
178/// [`PlanTraceObject::trace_object`].  It is applicable to all plans that implement
179/// [`PlanTraceObject`].
180///
181/// Plans usually don't implement `PlanTraceObject` directly, but use the
182/// `#[derive(PlanTraceObject)]` macro. See [`PlanTraceObject`] for more details.
183pub struct PlanTrace<P: Plan + PlanTraceObject<P::VM>, const KIND: TraceKind> {
184    plan: &'static P,
185}
186
187impl<P: Plan + PlanTraceObject<P::VM>, const KIND: TraceKind> Clone for PlanTrace<P, KIND> {
188    fn clone(&self) -> Self {
189        Self { plan: self.plan }
190    }
191}

193impl<P: Plan + PlanTraceObject<P::VM>, const KIND: TraceKind> Trace for PlanTrace<P, KIND> {
194    type VM = P::VM;
195
196    fn from_mmtk(mmtk: &'static MMTK<Self::VM>) -> Self {
197        let plan = mmtk.get_plan().downcast_ref::<P>().unwrap();
198        Self { plan }
199    }
200
201    fn trace_object<Q: ObjectQueue>(
202        &self,
203        worker: &mut GCWorker<Self::VM>,
204        object: ObjectReference,
205        queue: &mut Q,
206    ) -> ObjectReference {
207        self.plan.trace_object::<Q, KIND>(queue, object, worker)
208    }
209
210    fn post_scan_object(&self, object: ObjectReference) {
211        self.plan.post_scan_object(object);
212    }
213
214    fn may_move_objects() -> bool {
215        P::may_move_objects::<KIND>()
216    }
217}

259/// This trait represents an object queue to enqueue objects during tracing.
260pub trait ObjectQueue {
261    /// Enqueue an object into the queue.
262    fn enqueue(&mut self, object: ObjectReference);
263}
264
265impl<F: FnMut(ObjectReference)> ObjectQueue for F {
266    fn enqueue(&mut self, object: ObjectReference) {
267        self(object)
268    }
269}
270
271impl ObjectQueue for Option<ObjectReference> {
272    fn enqueue(&mut self, object: ObjectReference) {
273        debug_assert!(self.is_none());
274        *self = Some(object);
275    }
276}

285/// An implementation of `ObjectQueue` using a `Vec`.
286///
287/// This can also be used as a buffer. For example, the mark stack or the write barrier mod-buffer.
288pub struct VectorQueue<T> {
289    /// Enqueued nodes.
290    buffer: Vec<T>,
291}
292
293impl<T> VectorQueue<T> {
294    /// Reserve a capacity of this on first enqueue to avoid frequent resizing.
295    const CAPACITY: usize = EDGES_WORK_BUFFER_SIZE;
296
297    /// Create an empty `VectorObjectQueue`.
298    pub fn new() -> Self {
299        Self { buffer: Vec::new() }
300    }
301
302    /// Return `true` if the queue is empty.
303    pub fn is_empty(&self) -> bool {
304        self.buffer.is_empty()
305    }
306
307    /// Return the contents of the underlying vector.  It will empty the queue.
308    pub fn take(&mut self) -> Vec<T> {
309        std::mem::take(&mut self.buffer)
310    }
311
312    /// Consume this `VectorObjectQueue` and return its underlying vector.
313    pub fn into_vec(self) -> Vec<T> {
314        self.buffer
315    }
316
317    /// Check if the buffer size reaches `CAPACITY`.
318    pub fn is_full(&self) -> bool {
319        self.buffer.len() >= Self::CAPACITY
320    }
321
322    /// Push an element to the queue. If the queue is empty, it will reserve
323    /// space to hold the number of elements defined by the capacity.
324    /// The user of this method needs to make sure the queue length does
325    /// not exceed the capacity to avoid allocating more space
326    /// (this method will not check the length against the capacity).
327    pub fn push(&mut self, v: T) {
328        if self.buffer.is_empty() {
329            self.buffer.reserve(Self::CAPACITY);
330        }
331        self.buffer.push(v);
332    }
333
334    /// Return the len of the queue
335    pub fn len(&self) -> usize {
336        self.buffer.len()
337    }
338
339    /// Empty the queue
340    pub fn clear(&mut self) {
341        self.buffer.clear()
342    }
343}
344
345impl<T> Default for VectorQueue<T> {
346    fn default() -> Self {
347        Self::new()
348    }
349}
350
351impl ObjectQueue for VectorQueue<ObjectReference> {
352    fn enqueue(&mut self, v: ObjectReference) {
353        self.push(v);
354    }
355}

357/// For iterating over the slots of an object.
358// FIXME: This type iterates slots, but all of its current use cases only care about the values in the slots.
359// And it currently only works if the object supports slot enqueuing (i.e. `Scanning::scan_object` is implemented).

364
365impl<VM: VMBinding> SlotIterator<VM> {
366    /// Iterate over the slots of an object by applying a function to each slot.
367    pub fn iterate_fields<F: FnMut(VM::VMSlot)>(object: ObjectReference, _tls: VMThread, mut f: F) {
368        // FIXME: We should use tls from the arguments.
369        // See https://github.com/mmtk/mmtk-core/issues/1375
370        let fake_tls = VMWorkerThread(VMThread::UNINITIALIZED);
371        if !<VM::VMScanning as Scanning<VM>>::support_slot_enqueuing(fake_tls, object) {
372            panic!("SlotIterator::iterate_fields cannot be used on objects that don't support slot-enqueuing");
373        }
374        <VM::VMScanning as Scanning<VM>>::scan_object(fake_tls, object, &mut f);
375    }
