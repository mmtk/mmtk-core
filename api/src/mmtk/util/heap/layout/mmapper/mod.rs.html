9/// An `Mmapper` manages the mmap state of memory used by the heap and side metadata of MMTk.
10///
11/// For the efficiency of implementation, an `Mmapper` operates at the granularity of
12/// [`Mmapper::granularity()`].  Methods that take memory ranges as arguments will round the range
13/// to the overlapping chunks.
14///
15/// From the perspective of the `Mmapper`, each memory range can be in one of the three states:
16/// Unmapped, Quarantined, and Mapped.  The state transition graph is:
17///
18/// ```text
19/// ┌────────┐  ensure_mapped / mark_as_mapped    ┌──────┐
20/// │Unmapped├────────────────────────────────────►Mapped│
21/// └───┬────┘                                    └───▲──┘
22///     │               ┌───────────┐                 │
23///     └───────────────►Quarantined├─────────────────┘
24///         quarantine  └───────────┘  ensure_mapped
25/// ```
26///
27/// -   **Unmapped** means the memory is not mapped by the `Mmapper`, and may be mapped by other
28///     components of the process.
29/// -   **Quarantined** means the `Mmapper` has reserved the memory for MMTk, usually by using
30///     `mmap` with `PROT_NONE`.
31/// -   **Mapped** means `Mmapper` has mapped the memory, and the memory can be read and written by
32///     MMTk.
33pub trait Mmapper: Sync {
34    /// The logarithm of granularity of this `Mmapper`, in bytes.  Must be at least
35    /// [`LOG_BYTES_IN_PAGE`].
36    ///
37    /// See trait-level doc for [`Mmapper`] for details.
38    fn log_granularity(&self) -> u8;
39
40    /// The logarithm of the address space size this `Mmapper` can handle.
41    ///
42    /// In other words, this `Mmapper` cannot handle addresses greater than or equal to
43    /// `1 << self.log_mappable_bytes()`.
44    fn log_mappable_bytes(&self) -> u8;
45
46    /// The granularity of `Mmapper`.  Don't override this method.  Override
47    /// [`Mmapper::log_granularity`] instead.
48    ///
49    /// See trait-level doc for [`Mmapper`] for details.
50    fn granularity(&self) -> usize {
51        1 << self.log_granularity()
52    }
53
54    /// Mark a number of pages as mapped, without making any
55    /// request to the operating system.  Used to mark pages
56    /// that the VM has already mapped.
57    ///
58    /// Arguments:
59    /// * `start`: Address of the first page to be mapped
60    /// * `bytes`: Number of bytes to ensure mapped
61    fn mark_as_mapped(&self, start: Address, bytes: usize);
62
63    /// Quarantine/reserve address range. We mmap from the OS with no reserve and with PROT_NONE,
64    /// which should be little overhead. This ensures that we can reserve certain address range that
65    /// we can use if needed. Quarantined memory needs to be mapped before it can be used.
66    ///
67    /// Arguments:
68    /// * `start`: Address of the first page to be quarantined
69    /// * `pages`: Number of pages to quarantine from the start
70    /// * `strategy`: The mmap strategy.  The `prot` field is ignored because we always use
71    ///   `PROT_NONE`.
72    /// * `anno`: Human-readable annotation to apply to newly mapped memory ranges.
73    fn quarantine_address_range(
74        &self,
75        start: Address,
76        pages: usize,
77        huge_page_option: HugePageSupport,
78        anno: &MmapAnnotation,
79    ) -> Result<()>;
80
81    /// Ensure that a range of pages is mmapped (or equivalent).  If the
82    /// pages are not yet mapped, demand-zero map them. Note that mapping
83    /// occurs at chunk granularity, not page granularity.
84    ///
85    /// Arguments:
86    /// * `start`: The start of the range to be mapped.
87    /// * `pages`: The size of the range to be mapped, in pages
88    /// * `strategy`: The mmap strategy.
89    /// * `anno`: Human-readable annotation to apply to newly mapped memory ranges.
90    // NOTE: There is a monotonicity assumption so that only updates require lock
91    // acquisition.
92    // TODO: Fix the above to support unmapping.
93    fn ensure_mapped(
94        &self,
95        start: Address,
96        pages: usize,
97        huge_page_option: HugePageSupport,
98        prot: MmapProtection,
99        anno: &MmapAnnotation,
100    ) -> Result<()>;
101
102    /// Is the page pointed to by this address mapped? Returns true if
103    /// the page at the given address is mapped.
104    ///
105    /// Arguments:
106    /// * `addr`: Address in question
107    fn is_mapped_address(&self, addr: Address) -> bool;
108}
